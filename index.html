-- === CONFIGURAÇÃO === local milhoItem = "milho" -- Item que o jogador recebe ao colher local quantidadePorColheita = 8 -- Quantidade de milho por planta local tempoCrescimento = 60 -- Tempo em segundos para crescer (60 = 1 minuto) ESX = exports["es_extended"]:getSharedObject() -- ou QBCore se for QBCore -- Coordenadas das plantas (você pode adicionar várias) local plantasMilho = { vector3(1234.56, -4321.09, 30.0), vector3(1240.12, -4318.45, 30.0), -- Adicione mais aqui... } local plantasPlantadas = {} -- Armazena plantas plantadas -- ===================== COMANDO PARA PLANTAR ===================== RegisterCommand('plantarmilho', function(source) local xPlayer = ESX.GetPlayerFromId(source) local ped = GetPlayerPed(source) local coords = GetEntityCoords(ped) for _, plantCoord in pairs(plantasMilho) do if #(coords - plantCoord) < 3.0 then if xPlayer.getInventoryItem('semente_milho').count >= 1 then xPlayer.removeInventoryItem('semente_milho', 1) plantasPlantadas[plantCoord] = { tempoPlantado = GetGameTimer(), coords = plantCoord } TriggerClientEvent('ox_lib:notify', source, { title = 'Plantação', description = 'Milho plantado! Espere crescer...', type = 'success' }) return else TriggerClientEvent('ox_lib:notify', source, {description = 'Você não tem semente de milho!', type = 'error'}) return end end end end) -- ===================== COLHER ===================== RegisterNetEvent('colherMilho') AddEventHandler('colherMilho', function(plantCoord) local source = source local xPlayer = ESX.GetPlayerFromId(source) local plant = plantasPlantadas[plantCoord] if plant and (GetGameTimer() - plant.tempoPlantado) >= (tempoCrescimento * 1000) then xPlayer.addInventoryItem(milhoItem, quantidadePorColheita) TriggerClientEvent('ox_lib:notify', source, { title = 'Colheita', description = 'Você colheu '..quantidadePorColheita..' milhos!', type = 'success' }) plantasPlantadas[plantCoord] = nil -- Remove a planta else TriggerClientEvent('ox_lib:notify', source, {description = 'O milho ainda não cresceu!', type = 'error'}) end end) -- ===================== VENDER ===================== RegisterNetEvent('venderMilho') AddEventHandler('venderMilho', function() local source = source local xPlayer = ESX.GetPlayerFromId(source) local qtdMilho = xPlayer.getInventoryItem(milhoItem).count if qtdMilho > 0 then local pagamento = qtdMilho * 12 -- R$12 por milho xPlayer.removeInventoryItem(milhoItem, qtdMilho) xPlayer.addMoney(pagamento) TriggerClientEvent('ox_lib:notify', source, { title = 'Venda', description = 'Você vendeu '..qtdMilho..' milhos por R$'..pagamento, type = 'success' }) else TriggerClientEvent('ox_lib:notify', source, {description = 'Você não tem milho!', type = 'error'}) end end)
